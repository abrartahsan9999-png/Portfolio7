Customer Billing System (C)

A C project using file handling to manage customer billing operations like add, view, search, delete.

🔍 View Code 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Bill {
    int id;
    char name[50];
    float amount;
    char date[20];
};

void addBill() {
    FILE *fp = fopen("bills.dat", "ab");
    struct Bill b;
    if (fp == NULL) { printf("File could not be opened!\n"); return; }
    printf("\nEnter Bill ID: "); scanf("%d", &b.id); getchar();
    printf("Enter Customer Name: "); fgets(b.name, sizeof(b.name), stdin);
    b.name[strcspn(b.name, "\n")] = '\0';
    printf("Enter Amount: "); scanf("%f", &b.amount);
    printf("Enter Date (DD-MM-YYYY): "); scanf("%s", b.date);
    fwrite(&b, sizeof(b), 1, fp); fclose(fp);
    printf("Bill added successfully!\n\n");
}

void viewBills() {
    FILE *fp = fopen("bills.dat", "rb");
    struct Bill b; int found = 0;
    if (fp == NULL) { printf("No bills found.\n"); return; }
    printf("\n===== All Bills =====\n");
    while (fread(&b, sizeof(b), 1, fp)) {
        printf("ID: %d | Name: %s | Amount: %.2f | Date: %s\n", b.id, b.name, b.amount, b.date);
        found = 1;
    }
    fclose(fp);
    if (!found) printf("No records to display.\n");
    printf("======================\n\n");
}

void searchBill() {
    FILE *fp = fopen("bills.dat", "rb");
    struct Bill b; char searchName[50]; int found = 0;
    if (fp == NULL) { printf("No bills found.\n"); return; }
    getchar();
    printf("Enter customer name to search: ");
    fgets(searchName, sizeof(searchName), stdin);
    searchName[strcspn(searchName, "\n")] = '\0';
    while (fread(&b, sizeof(b), 1, fp)) {
        if (strcmp(b.name, searchName) == 0) {
            printf("Bill Found:\nID: %d | Name: %s | Amount: %.2f | Date: %s\n", b.id, b.name, b.amount, b.date);
            found = 1; break;
        }
    }
    fclose(fp);
    if (!found) printf("Bill not found.\n");
}

void deleteBill() {
    FILE *fp = fopen("bills.dat", "rb");
    FILE *temp = fopen("temp.dat", "wb");
    struct Bill b; int delID, found = 0;
    if (fp == NULL || temp == NULL) { printf("File error!\n"); return; }
    printf("Enter Bill ID to delete: "); scanf("%d", &delID);
    while (fread(&b, sizeof(b), 1, fp)) {
        if (b.id != delID) fwrite(&b, sizeof(b), 1, temp);
        else found = 1;
    }
    fclose(fp); fclose(temp);
    remove("bills.dat"); rename("temp.dat", "bills.dat");
    if (found) printf("Bill deleted successfully.\n\n");
    else printf("Bill ID not found.\n\n");
}

int main() {
    int choice;
    while (1) {
        printf("======== Customer Billing System ========\n");
        printf("1. Add New Bill\n2. View All Bills\n3. Search Bill by Name\n4. Delete Bill by ID\n5. Exit\n");
        printf("Choose an option: "); scanf("%d", &choice);
        switch (choice) {
            case 1: addBill(); break;
            case 2: viewBills(); break;
            case 3: searchBill(); break;
            case 4: deleteBill(); break;
            case 5: printf("Exiting. Thank you!\n"); exit(0);
            default: printf("Invalid choice.\n");
        }
    }
    return 0;
}

To-Do List App (Tkinter)

A simple To-Do List application built using Python's Tkinter library, featuring task addition, deletion, and a customizable dark theme.

🔍 View Code 

import tkinter as tk

root = tk.Tk()
root.title("To-Do List App")
root.geometry("400x400")
root.configure(bg="black")

tasks = []

entry = tk.Entry(root, font=("Arial", 14), bg="black", fg="white", insertbackground="lawngreen")
entry.pack(pady=10)

def add_task():
    task = entry.get()
    if task != "":
        tasks.append(task)
        listbox.insert(tk.END, task)
        entry.delete(0, tk.END)

def delete_task():
    selected = listbox.curselection()
    if selected:
        index = selected[0]
        tasks.pop(index)
        listbox.delete(index)

add_button = tk.Button(root, text="Add Task", width=15, command=add_task,
    bg="lawngreen", fg="black", activebackground="forestgreen",
    activeforeground="white", font=("Arial", 10, "bold"))
add_button.pack(pady=5)

listbox = tk.Listbox(root, font=("Arial", 14), width=30, height=10,
    bg="black", fg="white", selectbackground="lawngreen",
    selectforeground="black", borderwidth=2, relief="solid")
listbox.pack(pady=10)

delete_button = tk.Button(root, text="Delete Task", width=15, command=delete_task,
    bg="lawngreen", fg="black", activebackground="forestgreen",
    activeforeground="white", font=("Arial", 10, "bold"))
delete_button.pack(pady=5)

root.mainloop()
